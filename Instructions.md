# install vue-cli
$ npm install --global vue-cli

# create a new project using the "webpack" template
$ vue init webpack todo-app

# install dependencies and go!
$ cd todo-app
$ npm install

# to serve the app, run
npm run dev

# open browser and direct to:
http://localhost:8080
